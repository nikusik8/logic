function generateTruthTable() {
  const op = document.getElementById('logicOp').value;
  const ops = {
    AND: (a,b) => a && b,
    OR: (a,b) => a || b,
    NOT: (a) => !a,
    IMPLY: (a,b) => !a || b,
    EQUIV: (a,b) => a === b,
    XOR: (a,b) => a !== b,
    NOR: (a,b) => !(a || b),
    NAND: (a,b) => !(a && b)
  };
  
  let html = '<table>';
  if (op === 'NOT') {
    html += '<tr><th>A</th><th>NOT A</th></tr>';
    [false, true].forEach(a => {
      const res = ops[op](a);
      html += `<tr><td>${a}</td><td>${res}</td></tr>`;
    });
  } else {
    html += `<tr><th>A</th><th>B</th><th>A ${op} B</th></tr>`;
    [false, true].forEach(a => {
      [false, true].forEach(b => {
        const res = ops[op](a, b);
        html += `<tr><td>${a}</td><td>${b}</td><td>${res}</td></tr>`;
      });
    });
  }
  html += '</table>';
  document.getElementById('tableContainer').innerHTML = html;
}

        const sortingMethods = {
            insertionSort: function(arr) {

            bubbleSort: function(arr) {
                const array = [...arr];
                console.log('–ù–∞—á–∞–ª–æ –ø—É–∑—ã—Ä—å–∫–æ–≤–æ–π —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏');

            selectionSort: function(arr) {

            quickSort: function(arr) {
                if (arr.length <= 1) return arr;
                const pivot = arr[Math.floor(arr.length / 2)];
                const left = arr.filter(x => x < pivot);
                const right = arr.filter(x => x > pivot);
                const equal = arr.filter(x => x === pivot);
                return [...this.quickSort(left), ...equal, ...this.quickSort(right)];

            mergeSort: function(arr) {

                const left = this.mergeSort(arr.slice(0, mid));

            merge: function(left, right) {
                const result = [];
                while (left.length && right.length) {
                    result.push(left[0] < right[0] ? left.shift() : right.shift());

                return [...result, ...left, ...right];
